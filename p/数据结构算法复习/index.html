int Inverse_Link(LinkList *L)
{
    LinkList *cur, *p;
    if (L->next && L->next->next)
    {
        p = L->next->next;
        L->next->next = NULL;
        while (p)
        {
            cur = L->next;
            L->next = p;

            L->next->next = cur;

}

    p->prior = q;        //1<-->q<-->p<-->2
    return 0;
}

bool isPalindrome(struct ListNode *head)
{
    Lstack *s;
    int e;
    struct ListNode *p = head;
    s = (Lstack *)malloc(sizeof(Lstack));
    Initlist_Stack(s);
    while (p)
    {
        push(s, p->val);
        p = p->next;
    }
    p = head;
    while (p)
    {
        e = pop(s);
        if (p->val != e)
            break;
        p = p->next;
    }
    if (p)
        return false;
    else
        return true;
}
